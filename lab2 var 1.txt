#!/bin/bash
if [ -n "$1" ]
then
factorial=1
var1=$1
case $1 in
1|2|3|4|5|6|7|8|9)
while [ $var1 -gt 0 ]
do
factorial=$(($factorial*$var1))
var1=$(($var1 - 1))
done
echo Factorial of $1 is $factorial
exit 0;;
0) echo "Factorial of" $1 "is" $factorial
exit 0;;
*) echo "Invalid argument. Argument can contain only 0-9 numbers"
exit 1;;
esac
else
echo "Error: no argument"
exit 1
fi
exit 0
